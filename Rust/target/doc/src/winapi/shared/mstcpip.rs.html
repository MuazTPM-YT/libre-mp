351#[inline]
352pub fn IN4_IS_ADDR_RFC1918(a: &IN_ADDR) -> bool {
353    let s_addr = unsafe { *a.S_un.S_addr() };
354    ((s_addr & 0x00ff) == 0x0a) || ((s_addr & 0xf0ff) == 0x10ac) || ((s_addr & 0xffff) == 0xa8c0)
355}

372}
