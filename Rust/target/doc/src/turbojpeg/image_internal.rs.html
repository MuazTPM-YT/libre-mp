4/// An image with pixels of type `T`.
5///
6/// Three variants of this type are commonly used:
7///
8/// - `Image<&[u8]>`: immutable reference to image data (input image for compression by
9/// [`Compressor`][crate::Compressor])
10/// - `Image<&mut [u8]>`: mutable reference to image data (output image for decompression by
11/// [`Decompressor`][crate::Decompressor]).
12/// - `Image<Vec<u8>>`: owned image data (you can convert it to a reference using
13/// [`.as_deref()`][Image::as_deref] or [`.as_deref_mut()`][Image::as_deref_mut]).
14///
15/// Data for pixel in column `x` and row `y` is stored in `pixels` at offset `y*pitch +
16/// x*format.size()`.
17#[derive(Debug, Copy, Clone)]
18pub struct Image<T> {
19    /// Pixel data of the image (typically `&[u8]`, `&mut [u8]` or `Vec<u8>`).
20    pub pixels: T,
21    /// Width of the image in pixels (number of columns).
22    pub width: usize,
23    /// Pitch (stride) defines the size of one image row in bytes. Overlapping rows are not
24    /// supported, we require that `pitch >= width * format.size()`.
25    pub pitch: usize,
26    /// Height of the image in pixels (number of rows).
27    pub height: usize,
28    /// Format of pixels in memory, determines the color format (RGB, RGBA, grayscale or CMYK) and
29    /// the memory layout (RGB, BGR, RGBA, ...).
30    pub format: PixelFormat,
31}
32
33impl<T> Image<T> {
34    /// Converts from `&Image<T>` to `Image<&T::Target>`.
35    ///
36    /// In particular, you can use this to get `Image<&[u8]>` from `Image<Vec<u8>>`.
37    pub fn as_deref(&self) -> Image<&T::Target> where T: Deref {
38        Image {
39            pixels: self.pixels.deref(),
40            width: self.width,
41            pitch: self.pitch,
42            height: self.height,
43            format: self.format,
44        }
45    }
46
47    /// Converts from `&mut Image<T>` to `Image<&mut T::Target>`.
48    ///
49    /// In particular, you can use this to get `Image<&mut [u8]>` from `Image<Vec<u8>>`.
50    pub fn as_deref_mut(&mut self) -> Image<&mut T::Target> where T: DerefMut {
51        Image {
52            pixels: self.pixels.deref_mut(),
53            width: self.width,
54            pitch: self.pitch,
55            height: self.height,
56            format: self.format,
57        }
58    }
59
60    pub(crate) fn assert_valid(&self, pixels_len: usize) {
61        let Image { pixels: _, width, pitch, height, format } = *self;
62        assert!(pitch >= width*format.size(),
63            "pitch {} is too small for width {} and pixel format {:?}", pitch, width, format);
64        assert!(height == 0 || pitch*(height - 1) + width*format.size() <= pixels_len,
65            "pixels length {} is too small for width {}, height {}, pitch {} and pixel format {:?}",
66            pixels_len, width, height, pitch, format);
67    }
68}

70impl Image<Vec<u8>> {
71    /// Generates an image of the Mandelbrot set.
72    ///
73    /// The generated image has the given width and height and uses the given pixel format. This
74    /// method is intended for testing and demonstration purposes.
75    ///
76    /// # Example
77    ///
78    /// ```
79    /// let image = turbojpeg::Image::mandelbrot(200, 200, turbojpeg::PixelFormat::BGRA);
80    /// assert_eq!((image.width, image.height), (200, 200));
81    /// assert_eq!(image.format, turbojpeg::PixelFormat::BGRA);
82    /// ```
83    pub fn mandelbrot(width: usize, height: usize, format: PixelFormat) -> Image<Vec<u8>> {
84        // determine mapping from pixels to the complex plane
85
86        let radius = 2.;
87        let scale = usize::max(width, height) as f64 / (2. * radius);
88        let origin_x = width as f64 * 0.5;
89        let origin_y = height as f64 * 0.5;
90
91        let pixel_to_set = |pixel_x: usize, pixel_y: usize| -> (f64, f64) {
92            let (pixel_x, pixel_y) = (pixel_x as f64 + 0.5, pixel_y as f64 + 0.5);
93            let set_x = (pixel_x - origin_x) / scale;
94            let set_y = (pixel_y - origin_y) / scale;
95            (set_x, set_y)
96        };
97
98        // evaluate the mandelbrot set function
99
100        fn eval_set(set_x: f64, set_y: f64) -> f64 {
101            let max_iters = 100;
102            let (mut x, mut y) = (set_x, set_y);
103            let mut iters = 0;
104            while x*x + y*y <= 4. && iters < max_iters {
105                let next_x = x*x - y*y + set_x;
106                let next_y = 2.*x*y + set_y;
107                x = next_x;
108                y = next_y;
109                iters += 1;
110            }
111            1. - 0.99f64.powi(iters)
112        }
113
114        // convert the f64 values to pixel values
115
116        fn assign_rgba(r: usize, g: usize, b: usize, a: Option<usize>, data: &mut [u8], value: f64) {
117            data[b] = quantize(f64::clamp(f64::min(3.*value, 3. - 3.*value), 0., 1.));
118            data[r] = quantize(f64::clamp(f64::max(1. - 3.*value, 3.*value - 2.), 0., 1.));
119            data[g] = quantize(f64::clamp(value, 0., 1.));
120            if let Some(a) = a { data[a] = 255; }
121        }
122
123        fn assign_gray(data: &mut [u8], value: f64) {
124            data[0] = quantize(value);
125        }
126
127        fn quantize(value: f64) -> u8 {
128            (value * 255.) as u8
129        }
130
131
132        let pixel_size = format.size();
133        let assign_fn: &dyn Fn(&mut [u8], f64) = match format {
134            PixelFormat::RGB =>
135                &|data, value| assign_rgba(0,1,2,None, data, value),
136            PixelFormat::BGR =>
137                &|data, value| assign_rgba(2,1,0,None, data, value),
138            PixelFormat::RGBX | PixelFormat::RGBA =>
139                &|data, value| assign_rgba(0,1,2,Some(3), data, value),
140            PixelFormat::BGRX | PixelFormat::BGRA =>
141                &|data, value| assign_rgba(2,1,0,Some(3), data, value),
142            PixelFormat::XRGB | PixelFormat::ARGB =>
143                &|data, value| assign_rgba(1,2,3,Some(0), data, value),
144            PixelFormat::XBGR | PixelFormat::ABGR =>
145                &|data, value| assign_rgba(3,2,1,Some(0), data, value),
146            PixelFormat::GRAY =>
147                &assign_gray,
148            PixelFormat::CMYK =>
149                &|data, value| assign_rgba(0,1,2,Some(3), data, value),
150        };
151
152        // generate the image
153
154        let align = 32;
155        let pitch = (pixel_size * width + align - 1) / align * align;
156        let mut pixels = vec![0; pitch * height];
157
158        for y in 0..height {
159            for x in 0..width {
160                let (set_x, set_y) = pixel_to_set(x, y);
161                let value = eval_set(set_x, set_y);
162                assign_fn(&mut pixels[y*pitch + pixel_size*x..], value);
163            }
164        }
165
166        Image { pixels, width, pitch, height, format }
167    }
168}

170/// A YUV (YCbCr) planar image with pixels of type `T`.
171///
172/// This type stores an image in the JPEG color transform YCbCr (also called "YUV"). The image data
173/// first stores the Y plane, then the U (Cb) plane, and then the V (Cr) plane.
174///
175/// Two variants of this type are commonly used:
176///
177/// - `YuvImage<&mut [u8]>`: mutable reference to YUV image data (output image for decompression by
178/// [`Decompressor`][crate::Decompressor]).
179/// - `YuvImage<Vec<u8>>`: owned YUV image data (you can convert it to a reference using
180/// [`.as_deref()`][YuvImage::as_deref] or [`.as_deref_mut()`][YuvImage::as_deref_mut]).
181///
182/// # Image format
183///
184/// The size of each image plane is determined by the [width][Self::width], [height][Self::height],
185/// [chrominance subsampling][Self::subsamp] and [row alignment][Self::align] of the image:
186///
187/// - [Luminance (Y) plane width][Self::y_width()] is the image width padded to the nearest
188/// multiple of the [horizontal subsampling factor][Subsamp::width()].
189/// - [Luminance (Y) plane height][Self::y_height()] is the image height padded to the nearest
190/// multiple of the [vertical subsampling factor][Subsamp::height()].
191/// - [Chrominance (U and V) plane width][Self::uv_width()] is the luminance plane width divided by
192/// the horizontal subsampling factor.
193/// - [Chrominance (U and V) plane height][Self::uv_height()] is the luminance plane height divided
194/// by the vertical subsampling factor.
195/// - Each row is further padded to the nearest multiple of the [row alignment][Self::align].
196///
197/// ## Example
198///
199/// For example, if the source image is 35 x 35 pixels and [`Sub2x1`][Subsamp::Sub2x1] subsampling
200/// is used, then the luminance plane would be 36 x 35 bytes, and each of the chrominance planes
201/// would be 18 x 35 bytes. If you specify a row alignment of 4 bytes on top of this, then the
202/// luminance plane would be 36 x 35 bytes, and each of the chrominance planes would be 20 x 35
203/// bytes.
204///
205/// ```
206/// let img1 = turbojpeg::YuvImage {
207///     pixels: (),
208///     width: 35,
209///     align: 1,
210///     height: 35,
211///     subsamp: turbojpeg::Subsamp::Sub2x1,
212/// };
213/// assert_eq!(img1.y_size(), (36, 35));
214/// assert_eq!(img1.uv_size(), (18, 35));
215///
216/// let img2 = turbojpeg::YuvImage { align: 4, ..img1 };
217/// assert_eq!(img2.y_size(), (36, 35));
218/// assert_eq!(img2.uv_size(), (20, 35));
219/// ```
220pub struct YuvImage<T> {
221    /// Pixel data of the image (typically `&mut [u8]` or `Vec<u8>`).
222    pub pixels: T,
223    /// Width of the image in pixels (number of columns).
224    pub width: usize,
225    /// Row alignment (in bytes) of the YUV image (must be a power of 2). Each row in each plane of
226    /// the YUV image will be padded to the nearest multiple of `align`.
227    pub align: usize,
228    /// Height of the image in pixels (number of rows).
229    pub height: usize,
230    /// The level of chrominance subsampling used in the YUV image.
231    pub subsamp: Subsamp,
232}
233
234impl<T> YuvImage<T> {
235    /// Converts from `&YuvImage<T>` to `YuvImage<&T::Target>`.
236    ///
237    /// In particular, you can use this to get `YuvImage<&[u8]>` from `YuvImage<Vec<u8>>`.
238    pub fn as_deref(&self) -> YuvImage<&T::Target> where T: Deref {
239        YuvImage {
240            pixels: self.pixels.deref(),
241            width: self.width,
242            align: self.align,
243            height: self.height,
244            subsamp: self.subsamp,
245        }
246    }
247
248    /// Converts from `&mut YuvImage<T>` to `YuvImage<&mut T::Target>`.
249    ///
250    /// In particular, you can use this to get `YuvImage<&mut [u8]>` from `YuvImage<Vec<u8>>`.
251    pub fn as_deref_mut(&mut self) -> YuvImage<&mut T::Target> where T: DerefMut {
252        YuvImage {
253            pixels: self.pixels.deref_mut(),
254            width: self.width,
255            align: self.align,
256            height: self.height,
257            subsamp: self.subsamp,
258        }
259    }
260
261    /// Computes width of the luminance (Y) plane.
262    ///
263    /// This is the [image width][Self::width] padded to the nearest multiple of the [horizontal subsampling
264    /// factor][Subsamp::width()] and then aligned to the [row alignment][Self::align].
265    pub fn y_width(&self) -> usize {
266        let width = next_multiple_of(self.width, self.subsamp.width());
267        next_multiple_of(width, self.align)
268    }
269
270    /// Computes height of the luminance (Y) plane.
271    ///
272    /// This is the [image height][Self::height] padded to the nearest multiple of the [vertical
273    /// subsampling factor][Subsamp::height()].
274    pub fn y_height(&self) -> usize {
275        next_multiple_of(self.height, self.subsamp.height())
276    }
277
278    /// Computes size of the luminance (Y) plane.
279    pub fn y_size(&self) -> (usize, usize) {
280        (self.y_width(), self.y_height())
281    }
282
283    /// Computes width of each chrominance (U, V) plane.
284    ///
285    /// This is the [Y plane width][Self::y_width()] divided by the [horizontal subsampling
286    /// factor][Subsamp::width()] and then aligned to the [row alignment][Self::align].
287    pub fn uv_width(&self) -> usize {
288        let width = div_ceil(self.width, self.subsamp.width());
289        next_multiple_of(width, self.align)
290    }
291
292    /// Computes height of each chrominance (U, V) plane.
293    ///
294    /// This is the [Y plane height][Self::y_height()] divided by the [vertical subsampling
295    /// factor][Subsamp::height()].
296    pub fn uv_height(&self) -> usize {
297        div_ceil(self.height, self.subsamp.height())
298    }
299
300    /// Computes size of each chrominance (U, V) plane.
301    pub fn uv_size(&self) -> (usize, usize) {
302        (self.uv_width(), self.uv_height())
303    }
304
305    pub(crate) fn assert_valid(&self, pixels_len: usize) {
306        let YuvImage { pixels: _, width, align, height, subsamp } = *self;
307        let min_yuv_pixels_len = yuv_pixels_len(width, align, height, subsamp).unwrap();
308        assert!(min_yuv_pixels_len <= pixels_len,
309            "YUV pixels length {} is too small for width {}, height {}, align {} and subsamp {:?}",
310            pixels_len, width, height, align, subsamp);
311    }
312}

314/// Determine size in bytes of a YUV image.
315///
316/// Calculates the size for [`YuvImage::pixels`] based on the image width, height, chrominance
317/// subsampling and row alignment.
318///
319/// Returns an error on integer overflow. You can just `.unwrap()` the result if you don't care
320/// about this edge case.
321/// 
322/// # Example
323///
324/// ```
325/// // read JPEG data from file
326/// let jpeg_data = std::fs::read("examples/parrots.jpg")?;
327///
328/// // read the JPEG header
329/// let header = turbojpeg::read_header(&jpeg_data)?;
330/// // get YUV pixels length
331/// let align = 4;
332/// let yuv_pixels_len = turbojpeg::yuv_pixels_len(header.width, align, header.height, header.subsamp);
333/// assert_eq!(yuv_pixels_len.unwrap(), 294912);
334///
335/// # Ok::<(), Box<dyn std::error::Error>>(())
336/// ```
337#[doc(alias = "tj3YUVBufSize")]
338pub fn yuv_pixels_len(width: usize, align: usize, height: usize, subsamp: Subsamp) -> Result<usize> {
339    let width = width.try_into().map_err(|_| Error::IntegerOverflow("width"))?;
340    let align = align.try_into().map_err(|_| Error::IntegerOverflow("align"))?;
341    let height = height.try_into().map_err(|_| Error::IntegerOverflow("height"))?;
342    let len = unsafe { raw::tj3YUVBufSize(width, align, height, subsamp as libc::c_int) };
343    match len.try_into() {
344        Ok(0) => Err(Error::OutOfBounds),
345        Ok(len) => Ok(len),
346        Err(_) => Err(Error::IntegerOverflow("yuv size")),
347    }
348}

351/// A YUV (YCbCr) image with separate planes.
352///
353/// This type stores an image in the JPEG color transform YCbCr (also called "YUV") with separate
354/// Y, U (Cb), and V (Cr) planes, as opposed to interleaved YUV data like [`YuvImage`].
355///
356/// Each plane has its own data and stride for maximum flexibility.
357///
358/// Two variants of this type are commonly used:
359///
360/// - `YuvPlanesImage<&[u8]>`: immutable reference to YUV plane data (input for compression)
361/// - `YuvPlanesImage<Vec<u8>>`: owned YUV plane data
362#[derive(Debug, Copy, Clone)]
363pub struct YuvPlanesImage<T> {
364    /// Y (luminance) plane data.
365    pub y_plane: T,
366    /// U (chrominance) plane data.
367    pub u_plane: T,
368    /// V (chrominance) plane data.
369    pub v_plane: T,
370    /// Width of the image in pixels (number of columns).
371    pub width: usize,
372    /// Height of the image in pixels (number of rows).
373    pub height: usize,
374    /// Y plane stride (bytes per row).
375    pub y_stride: usize,
376    /// U plane stride (bytes per row).
377    pub u_stride: usize,
378    /// V plane stride (bytes per row).
379    pub v_stride: usize,
380    /// The level of chrominance subsampling used in the YUV image.
381    pub subsamp: Subsamp,
382}
383
384impl<T> YuvPlanesImage<T> {
385    /// Converts from `&YuvPlanesImage<T>` to `YuvPlanesImage<&T::Target>`.
386    pub fn as_deref(&self) -> YuvPlanesImage<&T::Target> where T: Deref {
387        YuvPlanesImage {
388            y_plane: self.y_plane.deref(),
389            u_plane: self.u_plane.deref(),
390            v_plane: self.v_plane.deref(),
391            width: self.width,
392            height: self.height,
393            y_stride: self.y_stride,
394            u_stride: self.u_stride,
395            v_stride: self.v_stride,
396            subsamp: self.subsamp,
397        }
398    }
399
400    /// Get a reference to a component plane.
401    pub fn plane(&self, component: YuvComponent) -> &T {
402        match component {
403            YuvComponent::Y => &self.y_plane,
404            YuvComponent::U => &self.u_plane,
405            YuvComponent::V => &self.v_plane,
406        }
407    }
408
409    /// Get a mutable reference to a component plane.
410    pub fn plane_mut(&mut self, component: YuvComponent) -> &mut T {
411        match component {
412            YuvComponent::Y => &mut self.y_plane,
413            YuvComponent::U => &mut self.u_plane,
414            YuvComponent::V => &mut self.v_plane,
415        }
416    }
417
418    /// Get the stride (bytes per row) of a component plane.
419    pub fn stride(&self, component: YuvComponent) -> usize {
420        match component {
421            YuvComponent::Y => self.y_stride,
422            YuvComponent::U => self.u_stride,
423            YuvComponent::V => self.v_stride,
424        }
425    }
426
427    pub(crate) fn assert_valid(&self, y_len: usize, u_len: usize, v_len: usize) {
428        let YuvPlanesImage { width, height, y_stride, u_stride, v_stride, subsamp, .. } = *self;
429
430        let min_y_plane_len = yuv_plane_len(YuvComponent::Y, width, y_stride, height, subsamp).unwrap();
431        assert!(min_y_plane_len <= y_len,
432            "Y plane length {} is too small for width {}, height {}, stride {} and subsamp {:?}",
433            y_len, width, height, y_stride, subsamp);
434
435        let min_u_plane_len = yuv_plane_len(YuvComponent::U, width, u_stride, height, subsamp).unwrap();
436        assert!(min_u_plane_len <= u_len,
437            "U plane length {} is too small for width {}, height {}, stride {} and subsamp {:?}",
438            u_len, width, height, u_stride, subsamp);
439
440        let min_v_plane_len = yuv_plane_len(YuvComponent::V, width, v_stride, height, subsamp).unwrap();
441        assert!(min_v_plane_len <= v_len,
442            "V plane length {} is too small for width {}, height {}, stride {} and subsamp {:?}",
443            v_len, width, height, v_stride, subsamp);
444    }
445}

455/// Determine size in bytes of a YUV plane.
456///
457/// Calculates the size for a plane in [`YuvPlanesImage`] based on the image width, height and
458/// chrominance subsampling.
459///
460/// Returns an error on integer overflow. You can just `.unwrap()` the result if you don't care
461/// about this edge case.
462#[doc(alias = "tj3YUVPlaneSize")]
463pub fn yuv_plane_len(
464    component: YuvComponent,
465    width: usize,
466    stride: usize,
467    height: usize,
468    subsamp: Subsamp,
469) -> Result<usize> {
470    let width = width.try_into().map_err(|_| Error::IntegerOverflow("width"))?;
471    let stride = stride.try_into().map_err(|_| Error::IntegerOverflow("stride"))?;
472    let height = height.try_into().map_err(|_| Error::IntegerOverflow("height"))?;
473    let len = unsafe {
474        raw::tj3YUVPlaneSize(
475            component as libc::c_int,
476            width, stride, height,
477            subsamp as libc::c_int,
478        )
479    };
480    match len.try_into() {
481        Ok(0) => Err(Error::OutOfBounds),
482        Ok(len) => Ok(len),
483        Err(_) => Err(Error::IntegerOverflow("yuv plane size")),
484    }
485}

489fn next_multiple_of(n: usize, divisor: usize) -> usize {
490    div_ceil(n, divisor) * divisor

493fn div_ceil(n: usize, divisor: usize) -> usize {
494    (n + divisor - 1) / divisor
