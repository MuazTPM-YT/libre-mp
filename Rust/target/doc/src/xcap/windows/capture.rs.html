8fn check_capture_region(
9    x: u32,
10    y: u32,
11    width: u32,
12    height: u32,
13    monitor_width: u32,
14    monitor_height: u32,
15) -> XCapResult<()> {
16    // Validate region bounds
17    if width > monitor_width
18        || height > monitor_height
19        || x + width > monitor_width
20        || y + height > monitor_height
21    {
22        return Err(XCapError::InvalidCaptureRegion(format!(
23            "Region ({x}, {y}, {width}, {height}) is outside monitor size ({monitor_width}, {monitor_height})"
24        )));
25    }
26    Ok(())
27}

29#[cfg(feature = "wgc")]
30pub(super) fn capture_monitor(
31    monitor: &ImplMonitor,
32    x: Option<u32>,
33    y: Option<u32>,
34    width: Option<u32>,
35    height: Option<u32>,
36) -> XCapResult<RgbaImage> {
37    use super::wgc;
38    if let (Some(x), Some(y), Some(width), Some(height)) = (x, y, width, height) {
39        let monitor_width = monitor.width()?;
40        let monitor_height = monitor.height()?;
41        check_capture_region(x, y, width, height, monitor_width, monitor_height)?;
42        wgc::capture_monitor(monitor.h_monitor, x, y, width, height)
43    } else {
44        wgc::capture_monitor(monitor.h_monitor, 0, 0, monitor.width()?, monitor.height()?)
45    }
46}

81#[cfg(feature = "wgc")]
82pub(super) fn capture_window(window: &ImplWindow) -> XCapResult<RgbaImage> {
83    use windows::Win32::System::Threading::{GetCurrentProcess, PROCESS_QUERY_LIMITED_INFORMATION};
84
85    use super::{
86        utils::{get_process_is_dpi_awareness, open_process},
87        wgc,
88    };
89
90    // 在win10之后，不同窗口有不同的dpi，所以可能存在截图不全或者截图有较大空白，实际窗口没有填充满图片
91    // 如果窗口不感知dpi，那么就不需要缩放，如果当前进程感知dpi，那么也不需要缩放
92    let scope_guard_handle = open_process(PROCESS_QUERY_LIMITED_INFORMATION, false, window.pid()?)?;
93    let window_is_dpi_awareness = get_process_is_dpi_awareness(*scope_guard_handle)?;
94    let current_process_is_dpi_awareness =
95        unsafe { get_process_is_dpi_awareness(GetCurrentProcess())? };
96
97    let scale_factor = if !window_is_dpi_awareness || current_process_is_dpi_awareness {
98        1.0
99    } else {
100        window.current_monitor()?.scale_factor()?
101    };
102    let width = (window.width()? as f32 * scale_factor).ceil() as u32;
103    let height = (window.height()? as f32 * scale_factor).ceil() as u32;
104    wgc::capture_window(window.hwnd, 0, 0, width, height)
105}
