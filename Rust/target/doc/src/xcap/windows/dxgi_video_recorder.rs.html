32pub(crate) struct ImplVideoRecorder {
33    d3d_device: ID3D11Device,
34    d3d_context: ID3D11DeviceContext,
35    duplication: IDXGIOutputDuplication,
36    recorder_waker: Arc<RecorderWaker>,
37    tx: SyncSender<Frame>,
38}
39
40impl ImplVideoRecorder {
41    pub fn new(h_monitor: HMONITOR) -> XCapResult<(Self, Receiver<Frame>)> {
42        unsafe {
43            let d3d_device = create_d3d_device(
44                D3D11_CREATE_DEVICE_BGRA_SUPPORT | D3D11_CREATE_DEVICE_SINGLETHREADED,
45            )?;
46            let dxgi_device = d3d_device.cast::<IDXGIDevice>()?;
47            let d3d_context = d3d_device.GetImmediateContext()?;
48
49            let adapter = dxgi_device.GetAdapter()?;
50
51            let mut output_index = 0;
52            loop {
53                let output = adapter.EnumOutputs(output_index)?;
54                output_index += 1;
55                let output_desc = output.GetDesc()?;
56
57                let output1 = output.cast::<IDXGIOutput1>()?;
58                let duplication = output1.DuplicateOutput(&dxgi_device)?;
59
60                if output_desc.Monitor == h_monitor {
61                    let (tx, sx) = sync_channel(0);
62                    let s = Self {
63                        d3d_device,
64                        d3d_context,
65                        duplication,
66                        recorder_waker: Arc::new(RecorderWaker::new()),
67                        tx,
68                    };
69                    s.on_frame()?;
70                    return Ok((s, sx));
71                }
72            }
73        }
74    }
75
76    pub fn on_frame(&self) -> XCapResult<()> {
77        let duplication = self.duplication.clone();
78        let d3d_device = self.d3d_device.clone();
79        let d3d_context = self.d3d_context.clone();
80        let recorder_waker = self.recorder_waker.clone();
81        let tx = self.tx.clone();
82
83        thread::spawn(move || {
84            loop {
85                recorder_waker.wait()?;
86
87                let mut frame_info = DXGI_OUTDUPL_FRAME_INFO::default();
88                let mut resource: Option<IDXGIResource> = None;
89                unsafe {
90                    match duplication.AcquireNextFrame(200, &mut frame_info, &mut resource) {
91                        Err(err) => {
92                            // 尝试释放当前帧，不然不能获取到下一帧数据
93                            let _ = duplication.ReleaseFrame();
94                            if err.code() != DXGI_ERROR_WAIT_TIMEOUT {
95                                break Err::<(), XCapError>(XCapError::new(
96                                    "DXGI_ERROR_UNSUPPORTED",
97                                ));
98                            }
99                        }
100                        _ => {
101                            // 如何确定 AcquireNextFrame 执行成功
102                            if frame_info.LastPresentTime != 0 {
103                                let resource =
104                                    resource.ok_or(XCapError::new("AcquireNextFrame failed"))?;
105                                let source_texture = resource.cast::<ID3D11Texture2D>()?;
106                                let mut source_texture_desc = D3D11_TEXTURE2D_DESC::default();
107                                source_texture.GetDesc(&mut source_texture_desc);
108
109                                let frame = texture_to_frame(
110                                    &d3d_device,
111                                    &d3d_context,
112                                    &source_texture,
113                                    0,
114                                    0,
115                                    source_texture_desc.Width,
116                                    source_texture_desc.Height,
117                                )?;
118                                let _ = tx.send(frame);
119                            }
120
121                            // 最后释放帧，不然获取不到当前帧的数据
122                            duplication.ReleaseFrame()?;
123                        }
124                    }
125                }
126            }
127        });
128
129        Ok(())
130    }
131
132    pub fn start(&self) -> XCapResult<()> {
133        self.recorder_waker.wake()?;
134
135        Ok(())
136    }
137
138    pub fn stop(&self) -> XCapResult<()> {
139        self.recorder_waker.sleep()?;
140
141        Ok(())
142    }
