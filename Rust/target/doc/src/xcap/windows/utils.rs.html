37pub(super) fn get_build_number() -> u32 {
38    unsafe {
39        let mut buf_len: u32 = 2048;
40        let mut buf: Vec<u16> = Vec::with_capacity(buf_len as usize);
41
42        let err = RegGetValueW(
43            HKEY_LOCAL_MACHINE,
44            w!(r"SOFTWARE\Microsoft\Windows NT\CurrentVersion"),
45            w!("CurrentBuildNumber"),
46            RRF_RT_REG_SZ,
47            None,
48            Some(buf.as_mut_ptr().cast()),
49            Some(&mut buf_len),
50        );
51
52        if err.is_err() {
53            return 0;
54        }
55
56        buf.set_len(buf_len as usize);
57
58        let build_version = U16CString::from_vec_truncate(buf)
59            .to_string()
60            .unwrap_or_default();
61
62        build_version.parse().unwrap_or(0)
63    }
64}

66pub(super) fn get_os_major_version() -> u8 {
67    let build_number = get_build_number();
68    // https://en.wikipedia.org/wiki/List_of_Microsoft_Windows_versions
69    if build_number >= 22000 {
70        11
71    } else if build_number >= 10240 {
72        10
73    } else if build_number >= 9200 {
74        8
75    } else {
76        7
77    }
78}

80pub(super) fn bgra_to_rgba(mut buffer: Vec<u8>) -> Vec<u8> {
81    let is_old_version = get_os_major_version() < 8;
82    for src in buffer.chunks_exact_mut(4) {
83        src.swap(0, 2);
84        // fix https://github.com/nashaofu/xcap/issues/92#issuecomment-1910014951
85        if src[3] == 0 && is_old_version {
86            src[3] = 255;
87        }
88    }
89
90    buffer
91}

97pub(super) fn get_process_is_dpi_awareness(process: HANDLE) -> XCapResult<bool> {
98    unsafe {
99        let scope_guard_hmodule = load_library(w!("Shcore.dll"))?;
100
101        let get_process_dpi_awareness_proc_address =
102            GetProcAddress(*scope_guard_hmodule, s!("GetProcessDpiAwareness")).ok_or(
103                XCapError::new("GetProcAddress GetProcessDpiAwareness failed"),
104            )?;
105
106        let get_process_dpi_awareness: GetProcessDpiAwareness =
107            mem::transmute(get_process_dpi_awareness_proc_address);
108
109        let mut process_dpi_awareness = 0;
110        // https://learn.microsoft.com/zh-cn/windows/win32/api/shellscalingapi/nf-shellscalingapi-getprocessdpiawareness
111        get_process_dpi_awareness(process, &mut process_dpi_awareness).ok()?;
112
113        // 当前进程不感知 DPI，则回退到 GetDeviceCaps 获取 DPI
114        Ok(process_dpi_awareness != 0)
115    }
116}

118pub(super) fn load_library(
119    lib_filename: PCWSTR,
120) -> XCapResult<ScopeGuard<HMODULE, impl FnOnce(HMODULE)>> {
121    unsafe {
122        let hmodule = LoadLibraryW(lib_filename)?;
123
124        if hmodule.is_invalid() {
125            return Err(XCapError::new(format!(
126                "LoadLibraryW error {:?}",
127                GetLastError()
128            )));
129        }
130
131        let scope_guard_hmodule = guard(hmodule, |val| {
132            if let Err(err) = FreeLibrary(val) {
133                log::error!("FreeLibrary {val:?} failed {err:?}");
134            }
135        });
136
137        Ok(scope_guard_hmodule)
138    }
139}

141pub(super) fn open_process(
142    dw_desired_access: PROCESS_ACCESS_RIGHTS,
143    b_inherit_handle: bool,
144    dw_process_id: u32,
145) -> XCapResult<ScopeGuard<HANDLE, impl FnOnce(HANDLE)>> {
146    unsafe {
147        let handle = OpenProcess(dw_desired_access, b_inherit_handle, dw_process_id)?;
148
149        if handle.is_invalid() {
150            return Err(XCapError::new(format!(
151                "OpenProcess error {:?}",
152                GetLastError()
153            )));
154        }
155
156        let scope_guard_handle = guard(handle, |val| {
157            if let Err(err) = CloseHandle(val) {
158                log::error!("CloseHandle {val:?} failed {err:?}");
159            }
160        });
161
162        Ok(scope_guard_handle)
163    }
164}

166pub(super) fn get_monitor_config(
167    monitor_info_ex_w: MONITORINFOEXW,
168) -> XCapResult<DISPLAYCONFIG_TARGET_DEVICE_NAME> {
169    unsafe {
170        let mut number_of_paths = 0;
171        let mut number_of_modes = 0;
172        GetDisplayConfigBufferSizes(
173            QDC_ONLY_ACTIVE_PATHS,
174            &mut number_of_paths,
175            &mut number_of_modes,
176        )
177        .ok()?;
178
179        let mut paths = vec![DISPLAYCONFIG_PATH_INFO::default(); number_of_paths as usize];
180        let mut modes = vec![DISPLAYCONFIG_MODE_INFO::default(); number_of_modes as usize];
181
182        QueryDisplayConfig(
183            QDC_ONLY_ACTIVE_PATHS,
184            &mut number_of_paths,
185            paths.as_mut_ptr(),
186            &mut number_of_modes,
187            modes.as_mut_ptr(),
188            None,
189        )
190        .ok()?;
191
192        for path in paths {
193            let mut source = DISPLAYCONFIG_SOURCE_DEVICE_NAME {
194                header: DISPLAYCONFIG_DEVICE_INFO_HEADER {
195                    r#type: DISPLAYCONFIG_DEVICE_INFO_GET_SOURCE_NAME,
196                    size: mem::size_of::<DISPLAYCONFIG_SOURCE_DEVICE_NAME>() as u32,
197                    adapterId: path.sourceInfo.adapterId,
198                    id: path.sourceInfo.id,
199                },
200                ..DISPLAYCONFIG_SOURCE_DEVICE_NAME::default()
201            };
202
203            if DisplayConfigGetDeviceInfo(&mut source.header) != 0 {
204                continue;
205            }
206
207            if source.viewGdiDeviceName != monitor_info_ex_w.szDevice {
208                continue;
209            }
210
211            let mut target = DISPLAYCONFIG_TARGET_DEVICE_NAME {
212                header: DISPLAYCONFIG_DEVICE_INFO_HEADER {
213                    r#type: DISPLAYCONFIG_DEVICE_INFO_GET_TARGET_NAME,
214                    size: mem::size_of::<DISPLAYCONFIG_TARGET_DEVICE_NAME>() as u32,
215                    adapterId: path.sourceInfo.adapterId,
216                    id: path.targetInfo.id,
217                },
218                ..DISPLAYCONFIG_TARGET_DEVICE_NAME::default()
219            };
220
221            if DisplayConfigGetDeviceInfo(&mut target.header) != 0 {
222                continue;
223            }
224
225            return Ok(target);
226        }
227
228        Err(XCapError::new("Get monitor name failed"))
229    }
230}

232/**
233 * 获取 window 可见的实际宽高，包含标题栏、边框等非客户区的部分
234 * 不同于 WINDOWINFO 中的 rcWindow 与 rcClient
235 *  - rcWindow 包含了窗口的标题栏和边框等非客户区的部分，与 GetWindowRect 获取的窗口位置和大小一致
236 *  - rcClient 只包含了窗口的客户区部分，与 GetClientRect 获取的窗口大小一致，但 GetClientRect 返回left 和 top 都是 0，因为客户区的坐标系是相对于窗口的客户区而言的
237 *
238 * 获取窗口真实 bounds 必须使用 DwmGetWindowAttribute，原因为：
239 * 自 Windows 10 起，桌面窗口管理器（DWM）会为部分窗口添加不可见的 resize 边框（通常 8px 左右），例如 chrome 浏览器的窗口，GetWindowRect 获取的窗口位置和大小包含了这个 resize 边框（eg返回值：-8，-8，2560, 1392）
240 */
241pub(super) fn get_window_bounds(hwnd: HWND) -> XCapResult<RECT> {
242    let mut rect = RECT::default();
243
244    unsafe {
245        DwmGetWindowAttribute(
246            hwnd,
247            DWMWA_EXTENDED_FRAME_BOUNDS,
248            &mut rect as *mut RECT as *mut c_void,
249            mem::size_of::<RECT>() as u32,
250        )?;
251    }
252
253    Ok(rect)
254}

277pub(super) fn texture_to_frame(
278    d3d_device: &ID3D11Device,
279    d3d_context: &ID3D11DeviceContext,
280    source_texture: &ID3D11Texture2D,
281    x: u32,
282    y: u32,
283    width: u32,
284    height: u32,
285) -> XCapResult<Frame> {
286    unsafe {
287        let mut src_desc = D3D11_TEXTURE2D_DESC::default();
288        source_texture.GetDesc(&mut src_desc);
289
290        // 边界检查（防止越界）
291        if x + width > src_desc.Width || y + height > src_desc.Height {
292            return Err(XCapError::new("ROI out of bounds"));
293        }
294
295        let staging_texture = {
296            let mut staging_desc = src_desc;
297            staging_desc.Width = width;
298            staging_desc.Height = height;
299            staging_desc.BindFlags = 0;
300            staging_desc.MiscFlags = 0;
301            staging_desc.Usage = D3D11_USAGE_STAGING;
302            staging_desc.CPUAccessFlags = D3D11_CPU_ACCESS_READ.0 as u32;
303
304            let mut staging = None;
305            d3d_device.CreateTexture2D(&staging_desc, None, Some(&mut staging))?;
306            staging.ok_or(XCapError::new("CreateTexture2D failed"))?
307        };
308
309        // GPU裁剪区域
310        let region = D3D11_BOX {
311            left: x,
312            top: y,
313            right: x + width,
314            bottom: y + height,
315            front: 0,
316            back: 1,
317        };
318
319        d3d_context.CopySubresourceRegion(
320            Some(&staging_texture.cast()?),
321            0,
322            0,
323            0,
324            0,
325            Some(&source_texture.cast()?),
326            0,
327            Some(&region),
328        );
329
330        let resource: ID3D11Resource = staging_texture.cast()?;
331        let mut mapped = D3D11_MAPPED_SUBRESOURCE::default();
332        d3d_context.Map(
333            Some(&resource.clone()),
334            0,
335            D3D11_MAP_READ,
336            0,
337            Some(&mut mapped),
338        )?;
339
340        let mut bgra = vec![0u8; (width * height * 4) as usize];
341        let src_ptr = mapped.pData as *const u8;
342
343        for row in 0..height {
344            let src_offset = (row * mapped.RowPitch) as usize;
345            let dst_offset = (row * width * 4) as usize;
346
347            let src_slice =
348                std::slice::from_raw_parts(src_ptr.add(src_offset), (width * 4) as usize);
349
350            bgra[dst_offset..dst_offset + (width * 4) as usize].copy_from_slice(src_slice);
351        }
352
353        d3d_context.Unmap(Some(&resource), 0);
354
355        Ok(Frame::new(width, height, bgra_to_rgba(bgra.to_owned())))
356    }
